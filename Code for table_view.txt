SELECT gp.date,
    gp.all_grades_all_formulations,
    gp.all_grades_conventional,
    gp.all_grades_reformulated,
    gp.regular_all_formulations,
    gp.regular_conventional,
    gp.regular_reformulated,
    gp.midgrade_all_formulations,
    gp.midgrade_conventional,
    gp.midgrade_reformulated,
    gp.premium_all_formulations,
    gp.premium_conventional,
    gp.premium_reformulated,
    gp.no_2_diesel,
    tsa.date_yyyyddmm,
    tsa.miles_traveled,
    tsa.spending_bridge,
    tsa.spending_pavement,
    tsa.spending_total,
    tsa.construction_costindex,
    tsa.diesel_price,
    tsa.gas_price,
    tsa.personal_transport_cost,
    tsa.autosales_millions
   FROM gas_prices gp
     JOIN transportation_statistics_aws tsa ON gp.date = tsa.date_yyyyddmm
